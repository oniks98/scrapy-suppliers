    """

    if not description_html:
        return ""
    
    from scrapy import Selector
    desc_selector = Selector(text=description_html)
    paragraphs = desc_selector.css("p")
    
    result_parts = []
    for p in paragraphs:
        # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –ø–∞—Ä–∞–≥—Ä–∞—Ñ —Å –∫–ª–∞—Å—Å–æ–º card-header__analog-link

        if inner_html:
            result_parts.append(inner_html)
    
    # –°–æ–µ–¥–∏–Ω—è–µ–º –ø–∞—Ä–∞–≥—Ä–∞—Ñ—ã —Å <br> –º–µ–∂–¥—É –Ω–∏–º–∏
